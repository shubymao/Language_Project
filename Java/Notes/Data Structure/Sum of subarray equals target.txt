Subarray to Target Sum


public int numsubarraysumtarget(int[] arr, int target){
	int res=0;
	HashMap<Integer,Integer> mp = new HashMap<>();
	mp.put(0,1);
	int sum = 0;
	for(int num:arr){
		sum+=num;
		res+=mp.getOrDefault(sum-target,0);
		mp.put(sum, mp.getOrDefault(sum,0));
	}
	return res;
		
}
